    <h1>🥃 Team Shot Counter</h1>

    <!-- Team hinzufügen -->
    <input type="text" id="teamInput" placeholder="Teamname eingeben">
    <button onclick="addTeam()">Team hinzufügen</button>

    <!-- Teams werden hier angezeigt -->
    <div id="teams"></div>

    <script>
        let teams = [];

        function addTeam() {
            const name = document.getElementById("teamInput").value.trim();
            if (name === "") return;

            teams.push({ name: name, score: 0 });
            document.getElementById("teamInput").value = "";
            renderTeams();
        }

        function addPoint(index) {
            teams[index].score++;
            renderTeams();
        }

        function removePoint(index) {
            if (teams[index].score > 0) {
                teams[index].score--;
            }
            renderTeams();
        }

        function renderTeams() {
            const container = document.getElementById("teams");
            container.innerHTML = "";

            teams.forEach((team, index) => {
                container.innerHTML += `
                    <div class="team">
                        <h2>${team.name}</h2>
                        <div class="score">${team.score}</div>
                        <button onclick="addPoint(${index})">+1</button>
                        <button onclick="removePoint(${index})">-1</button>
                    </div>
                `;
            });
        }
    </script>
